require_once(dirname(__FILE__) . '/../bootstrap.php');

use MCSDK\helpers\CaptureRequestTokenResponse;
use MCSDK\oidc\RequestTokenResponse;

class CaptureRequestTokenResponseTest extends PHPUnit_Framework_TestCase
{

    public function testCaptureRequestTokenResponseTest()
    {
        $newRequestTokenResponse = new RequestTokenResponse();
        $captureRequestTokenResponse = new CaptureRequestTokenResponse();

        $captureRequestTokenResponse->complete($newRequestTokenResponse);

        $this->assertEquals($newRequestTokenResponse, $captureRequestTokenResponse->getRequestTokenResponse());
    }

}
