namespace MCSDK\oidc;

/**
 * Class to hold the response from {@link 
 * IOIDC#parseAuthenticationResponse(String,
 * IParseAuthenticationResponseCallback)}.
 */
class ParsedAuthorizationResponse
{
    private $_error;
    private $_error_description;
    private $_error_uri;
    private $_state;
    private $_code;

    /**
     * Get the value of the error parameter.
     *
     * @return string The error value.
     */
    public function get_error()
    {
        return $this->_error;
    }

    /**
     * Set the value of the error parameter.
     *
     * @param string $error The error value.
     */
    public function set_error($error)
    {
        $this->_error = $error;
    }

    /**
     * The value of the error_description parameter.
     *
     * @return string The error_description value.
     */
    public function get_error_description()
    {
        return $this->_error_description;
    }

    /**
     * Set the value of the error_description parameter.
     *
     * @param string $error_description The error_description value.
     */
    public function set_error_description($error_description)
    {
        $this->_error_description = $error_description;
    }

    /**
     * Get the value of the error_uri parameter.
     *
     * @return string The error_uri value.
     */
    public function get_error_uri()
    {
        return $this->_error_uri;
    }

    /**
     * Set the value of the error_uri parameter.
     *
     * @param string $error_uri The error_uri value.
     */
    public function set_error_uri($error_uri)
    {
        $this->_error_uri = $error_uri;
    }

    /**
     * Get the value of the state parameter.
     *
     * @return string The state value.
     */
    public function get_state()
    {
        return $this->_state;
    }

    /**
     * Set the value of the state parameter.
     *
     * @param string $state The state value.
     */
    public function set_state($state)
    {
        $this->_state = $state;
    }

    /**
     * Get the value of the code parameter.
     *
     * @return string The code value.
     */
    public function get_code()
    {
        return $this->_code;
    }

    /**
     * Set the value of the code parameter.
     *
     * @param string $code The code value.
     */
    public function set_code($code)
    {
        $this->_code = $code;
    }
}
