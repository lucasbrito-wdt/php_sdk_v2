namespace MCSDK\utils;

/**
 * Exception thrown when an expected field is not found while parsing a Json object.
 */
class NoFieldException extends \Exception
{

    /**
     * Throw an exception if a field cannot be found with that name
     *
     * @param string $fieldName the name of the field causing the exception
     */
    public function NoFieldException($fieldName)
    {
        parent::__construct('no field found named "' . $fieldName . '".');
    }

}
